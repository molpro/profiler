Detailed Description

Framework for timing code sections.

Example of use:

USE profilerf
IMPLICIT NONE
TYPE(Profiler) :: p
INTEGER, PARAMETER :: repeat=20000000
DOUBLE PRECISION :: a
INTEGER :: i
p = profiler('Fortran')
CALL p%start('sqrt')
a=1d0
DO i=1,repeat
a=a*sqrt(a+i)/sqrt(a+i+1)
END DO
CALL p%stop('sqrt',2*repeat)
print *,a
CALL p%start('exp')
a=1d0
DO i=1,repeat
a=a*exp(a+1d0/i)/exp(a+1d0/i+1)
END DO
print *,a
CALL p%stop('exp',2*repeat)
CALL p%print(6)
